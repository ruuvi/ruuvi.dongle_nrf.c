289 rd_status_t app_ble_scan_start (void)
290 {
291  rd_status_t err_code = RD_SUCCESS;
292 
293  if (scan_is_enabled (&m_scan_params))
294  {
295  err_code |= rt_adv_uninit();
296  err_code |= ri_radio_uninit();
297  rt_adv_init_t adv_params =
298  {
299  .channels = m_scan_params.scan_channels,
300  .adv_interval_ms = (1000U),
301  .adv_pwr_dbm = (0),
302  .manufacturer_id = m_scan_params.manufacturer_id
303  };
304 
305  if (!m_scan_params.manufacturer_filter_enabled)
306  {
307  adv_params.manufacturer_id = RB_BLE_UNKNOWN_MANUFACTURER_ID;
308  }
309 
310  if (RD_SUCCESS == err_code)
311  {
312  next_modulation_select();
313  err_code |= pa_lna_ctrl();
314  err_code |= ri_radio_init (m_scan_params.current_modulation);
315 
316  if (RD_SUCCESS == err_code)
317  {
318  err_code |= rt_adv_init (&adv_params);
319  err_code |= rt_adv_scan_start (&on_scan_isr);
320  err_code |= ri_watchdog_feed();
321  }
322  }
323  }
324  else
325  {
326  err_code |= app_ble_scan_stop();
327  }
328 
329  return err_code;
330 }
